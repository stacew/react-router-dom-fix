## Install

```bash
npm install --save react-router-dom-fix
```

## Usage

```tsx
import React, { Component } from 'react'

import MyComponent from 'react-router-dom-fix'
import 'react-router-dom-fix/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
